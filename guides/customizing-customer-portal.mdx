The customer portal allows buyers to manage their subscriptions directly, including pausing, resuming, cancelling, and updating delivery preferences. This guide covers customization of the buyer experience.

## Overview

The customer portal is built as a Customer Account UI Extension:
- **Location**: `extensions/buyer-subscriptions/`
- **Framework**: React with Shopify UI Extensions
- **Entry Point**: Customer account page
- **Features**: View subscriptions, manage delivery, skip orders, cancel

## Extension Architecture

```typescript filename="extensions/buyer-subscriptions/src/App.tsx"
import {SubscriptionList} from './SubscriptionList';
```

## Subscription Details View

### Loading Subscription Data

```typescript filename="hooks/useSubscriptionContract.ts"
export function useSubscriptionContract(id: string) {
  const {extensionApi} = useExtensionApi();
  const [subscription, setSubscription] = useState(null);
  const [loading, setLoading] = useState(true);

  const fetchSubscription = useCallback(async () => {
    setLoading(true);
    
    const result = await extensionApi.query(
      SubscriptionDetailsQuery,
      {variables: {id}},
    );

    if (result.data?.subscriptionContract) {
      setSubscription(result.data.subscriptionContract);
    }
    
    setLoading(false);
  }, [id, extensionApi]);

  useEffect(() => {
    fetchSubscription();
  }, [fetchSubscription]);

  return {
    subscription,
    loading,
    refetch: fetchSubscription,
  };
}
```

## Customer Actions

### Pausing Subscriptions

```typescript filename="components/PauseSubscriptionModal/PauseSubscriptionModal.tsx"
export function PauseSubscriptionModal({
  open,
  subscriptionContractId,
  onClose,
}: Props) {
  const {extensionApi} = useExtensionApi();
  const [loading, setLoading] = useState(false);
  const {showSuccessToast} = useToast();

  const handlePause = useCallback(async () => {
    setLoading(true);

    const result = await extensionApi.query(
      PauseSubscriptionContractMutation,
      {
        variables: {subscriptionContractId},
      },
    );

    if (result.data?.subscriptionContractPause?.userErrors.length === 0) {
      showSuccessToast(SuccessToastType.SubscriptionPaused);
      onClose();
    } else {
      // Handle errors
    }

    setLoading(false);
  }, [subscriptionContractId, extensionApi]);

  return (
    <Modal
      id="pause-subscription-modal"
      title="Pause subscription"
      padding
      onClose={onClose}
    >
      <BlockStack spacing="base">
        <Text>Your subscription will be paused until you resume it.</Text>
        <InlineStack spacing="base">
          <Button kind="secondary" onPress={onClose}>
            Cancel
          </Button>
          <Button kind="primary" onPress={handlePause} loading={loading}>
            Pause subscription
          </Button>
        </InlineStack>
      </BlockStack>
    </Modal>
  );
}
```

### Cancelling Subscriptions

```typescript filename="components/CancelSubscriptionModal/CancelSubscriptionModal.tsx"
export function CancelSubscriptionModal(props: Props) {
  const {extensionApi} = useExtensionApi();
  const [loading, setLoading] = useState(false);
  const {showSuccessToast} = useToast();

  const handleCancel = useCallback(async () => {
    setLoading(true);

    const result = await extensionApi.query(
      CancelSubscriptionMutation,
      {
        variables: {subscriptionContractId: props.subscriptionContractId},
      },
    );

    const {userErrors} = result.data?.subscriptionContractCancel || {};

    if (userErrors?.length === 0) {
      showSuccessToast(SuccessToastType.SubscriptionCancelled);
      props.onSuccess();
    }

    setLoading(false);
  }, [props, extensionApi]);

  return (
    <Modal
      id="cancel-subscription-modal"
      title="Cancel subscription"
      padding
      onClose={props.onClose}
    >
      <BlockStack spacing="base">
        <Text>
          Are you sure you want to cancel? This action cannot be undone.
        </Text>
        <InlineStack spacing="base">
          <Button kind="secondary" onPress={props.onClose}>
            Keep subscription
          </Button>
          <Button 
            kind="primary" 
            onPress={handleCancel} 
            loading={loading}
            tone="critical"
          >
            Cancel subscription
          </Button>
        </InlineStack>
      </BlockStack>
    </Modal>
  );
}
```

### Skipping Orders

```typescript filename="components/SkipNextOrderModal/SkipNextOrderModal.tsx"
export function SkipNextOrderModal({open, billingCycle, onClose}: Props) {
  const {extensionApi} = useExtensionApi();
  const [loading, setLoading] = useState(false);

  const handleSkip = useCallback(async () => {
    setLoading(true);

      SkipBillingCycleMutation,

            Cancel
          </Button>
          <Button kind="primary" onPress={handleSkip} loading={loading}>
            Skip order
          </Button>
        </InlineStack>
      </BlockStack>

  );
}
```

## Updating Delivery Address

Customers can update their delivery address:

```typescript filename="components/DeliveryModal/DeliveryModal.tsx"
export function DeliveryModal({
  currentAddress,
  subscriptionContractId,
  onClose,
  refetchSubscriptionContract,
}: Props) {
  const [loading, setLoading] = useState(false);
  const [modalState, setModalState] = useState(DeliveryModalState.AddressInput);
  const [newAddress, setNewAddress] = useState<Address>(currentAddress);
  const [deliveryOptions, setDeliveryOptions] = useState<DeliveryOption[]>([]);
  const [token, setToken] = useState<string>();

  const getDeliveryOptions = useFetchDeliveryOptions();

  const handleAddressSave = useCallback(async () => {
    setLoading(true);

    // Fetch available delivery methods for new address
    const {
      deliveryOptions,
      deliveryOptionsToken,
      errors,
    } = await getDeliveryOptions(newAddress, subscriptionContractId);

    setLoading(false);

    if (errors.length === 0 && deliveryOptions.length > 0) {
      setDeliveryOptions(deliveryOptions);
      setToken(deliveryOptionsToken);
      setModalState(DeliveryModalState.DeliveryMethodSelect);
    }
  }, [getDeliveryOptions, newAddress, subscriptionContractId]);

  const modalContent = (() => {
    switch (modalState) {
      case DeliveryModalState.AddressInput:
        return (
          <AddressModal
            address={newAddress}
            onSave={handleAddressSave}
            onClose={onClose}
            loading={loading}
          />
        );
      case DeliveryModalState.DeliveryMethodSelect:
        return (
          <DeliveryMethodSelectModal
            address={newAddress}
            deliveryOptions={deliveryOptions}
            deliveryOptionsToken={token}
            subscriptionContractId={subscriptionContractId}
            onSuccess={() => {
              refetchSubscriptionContract();
              onClose();
            }}
          />
        );
    }
  })();

  return (
    <Modal id="delivery-modal" padding title="Update delivery" onClose={onClose}>
      {modalContent}
    </Modal>
  );
}
```

<Note>
  Address updates require fetching new delivery options and selecting a shipping method before saving.
</Note>

## API Communication

The extension uses the Customer Account API:

```typescript filename="foundation/Api/useExtensionApi.ts"
export function useExtensionApi() {
  const api = useApi();

  const query = useCallback(
    async (query: string, options: QueryOptions) => {
      try {
        const response = await api.query(query, options);
        return response;
      } catch (error) {
        console.error('API Error:', error);
        throw error;
      }
    },
    [api],
  );

  return {
    extensionApi: {query},
    i18n: useTranslate(),
  };
}
```

## Error Handling

```typescript filename="foundation/ErrorBoundary.tsx"
export class ErrorBoundary extends React.Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = {hasError: false};
  }

  static getDerivedStateFromError(error: Error) {
    return {hasError: true};
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    console.error('Extension error:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return (
        <BlockStack spacing="base">
          <Text size="large">Something went wrong</Text>
          <Button onPress={() => this.setState({hasError: false})}>
            Try again
          </Button>
        </BlockStack>
      );
    }

    return this.props.children;
  }
}
```

## Toast Notifications

```typescript filename="utilities/hooks/useToast.ts"
export enum SuccessToastType {
  SubscriptionPaused = 'subscription_paused',
  SubscriptionResumed = 'subscription_resumed',
  SubscriptionCancelled = 'subscription_cancelled',
  OrderSkipped = 'order_skipped',
  DeliveryUpdated = 'delivery_updated',
}

export function useToast() {
  const {extensionApi} = useExtensionApi();

  const showSuccessToast = useCallback(
    (type: SuccessToastType) => {
      extensionApi.toast.show(getToastMessage(type), {
        duration: 3000,
      });
    },
    [extensionApi],
  );

  return {showSuccessToast};
}
```

## Best Practices

<Steps>
  <Step title="Provide clear feedback">

  <Step title="Confirm destructive actions">

  <Step title="Handle errors gracefully">
    Display user-friendly error messages and offer retry options.
  </Step>

  <Step title="Optimize performance">
    Cache subscription data and only refetch when necessary.
