---
title: POS Extension
description: Point of Sale extension for adding subscriptions to in-person purchases
---

The POS extension enables merchants to add subscriptions to products during in-person sales at Point of Sale. It provides a tile on the POS home screen and line item actions for managing subscriptions.

## Extension Configuration

Configured in `shopify.extension.toml`:

```toml
api_version = "2025-10"

[[extensions]]
type = "ui_extension"
name = "pos-extension"
handle = "pos-extension"
description = "Subscriptions"

# Home screen tile
[[extensions.targeting]]
module = "./src/Tile.tsx"
target = "pos.home.tile.render"

# Home screen modal
[[extensions.targeting]]
module = "./src/TileModal.tsx"
target = "pos.home.modal.render"

# Line item action button
[[extensions.targeting]]
module = "./src/MenuItem.tsx"
target = "pos.cart.line-item-details.action.menu-item.render"

# Line item action modal
[[extensions.targeting]]
module = "./src/MenuItemModal.tsx"
target = "pos.cart.line-item-details.action.render"
```

## Subscription Screen

The main interface for selecting subscription plans:

```tsx
import '@shopify/ui-extensions/preact';

export const SubscriptionScreen = ({
  lineItem,
  cartApi,
  productSearch,
  session,
  i18n,
}: Props) => {
  const {
    sellingPlansState,
    variantState,
    selectedPlan,
    setSelectedPlan,
    handleApplyPress,
    oneTimePurchasePrice,
    fetchSellingPlans,
  } = useSubscriptionScreen(lineItem, cartApi, productSearch, session);

  return (
    <s-page>
      <s-button
        slot="secondary-actions"
        onClick={handleApplyPress}
        disabled={!isSuccess(sellingPlansState)}
      >
        {i18n.t('SubscriptionScreen.save')}
      </s-button>
      <s-scroll-box>
        <s-stack direction="block" paddingInline="large-100">
          <VariantInfo
            productName={getProductNameFromVariant(variantState.data, lineItem)}
            variantName={getVariantName(variantState.data)}
            image={getVariantImage(variantState.data)}
          />
          {!lineItem.requiresSellingPlan ? (
            <PlanItem
              title={i18n.t('SubscriptionScreen.oneTimePurchase')}
              subtitle={oneTimePurchasePrice}
              isSelected={selectedPlan === undefined}
              onSelect={() => setSelectedPlan(undefined)}
            />
          ) : null}
          {sellingPlansState.data.map((group, groupIndex) => (
            <s-section key={`group-${groupIndex}`} heading={`${group.name}`}>
              {group.plans.map((plan, planIndex) => (
                <PlanItem
                  key={planIndex}
                  title={plan.name}
                  subtitle={formatPricingPolicy(plan.pricingPolicy)}
                  isSelected={selectedPlan?.id === plan.id}
                  onSelect={() => setSelectedPlan(plan)}
                />
              ))}
            </s-section>
          ))}
        </s-stack>
      </s-scroll-box>
    </s-page>
  );
};
```

## Custom Hooks

### useSubscriptionScreen

Manages subscription screen state and logic:

```typescript
export function useSubscriptionScreen(
  lineItem: LineItem,
  cartApi: CartApiContent,
  productSearch: ProductSearchApiContent,
  session: SessionApiContent,
) {
  const [selectedPlan, setSelectedPlan] = useState<SellingPlan | undefined>();
  const [sellingPlansState, setSellingPlansState] = useState<AsyncState>();
  const [variantState, setVariantState] = useState<AsyncState>();

  const fetchSellingPlans = async () => {
    setSellingPlansState({state: 'loading'});
    
    try {
      const result = await query({
        query: SellingPlansQuery,
        variables: {id: lineItem.productVariantId},
      });
      
      setSellingPlansState({
        state: 'success',
        data: result.data.productVariant.sellingPlanGroups,
      });
    } catch (error) {
      setSellingPlansState({state: 'error', error});
    }
  };

  const handleApplyPress = async () => {
    if (selectedPlan) {
      await cartApi.applySellingPlanToLineItem(
        lineItem.id,
        selectedPlan.id
      );
    } else {
      await cartApi.removeSellingPlanFromLineItem(lineItem.id);
    }
    window.close();
  };

  return {
    sellingPlansState,
    variantState,
    selectedPlan,
    setSelectedPlan,
    handleApplyPress,
    oneTimePurchasePrice: formatPrice(lineItem.price),
    fetchSellingPlans,
  };
}
```

### useSellingPlanTile

Manages home tile state:

```typescript
export function useSellingPlanTile(
  cart: Cart,
  i18n: PreactI18n,
  posVersion: string,
) {
  const [enabled, setEnabled] = useState(false);
  const [subtitle, setSubtitle] = useState('');
  
  const eligibleItems = cart.lineItems.filter(
    (item) => item.hasSellingPlanGroups
  );
  
  const subscriptionItems = cart.lineItems.filter(
    (item) => item.sellingPlan !== null
  );

  const onCartChange = (newCart: Cart) => {
    const hasEligible = newCart.lineItems.some(
      (item) => item.hasSellingPlanGroups
    );
    setEnabled(hasEligible);
    
    const count = newCart.lineItems.filter(
      (item) => item.sellingPlan !== null
    ).length;
    setSubtitle(
      i18n.t('Tile.subtitle', {count})
    );
  };

  return {
    title: i18n.t('Tile.title'),
    subtitle,
    enabled,
    disabledReason: enabled ? null : 'no-eligible-items',
    onCartChange,
  };
}
```

### useSellingPlans

Fetches selling plans via GraphQL:

```typescript
export function useSellingPlans(variantId: string) {
  const {query} = useGraphqlQuery();
  
  const [state, setState] = useState<AsyncState>({
    state: 'loading',
  });

  useEffect(() => {
    async function fetchPlans() {
      try {
        const result = await query({
          query: SellingPlansQuery,
          variables: {id: variantId},
        });
        
        const sellingPlanGroups = 
          result.data.productVariant.sellingPlanGroups.edges.map(
            (edge) => ({
              id: edge.node.id,
              name: edge.node.name,
              plans: edge.node.sellingPlans.edges.map(
                (planEdge) => ({
                  id: planEdge.node.id,
                  name: planEdge.node.name,
                  pricingPolicy: planEdge.node.pricingPolicies[0],
                })
              ),
            })
          );
        
        setState({state: 'success', data: sellingPlanGroups});
      } catch (error) {
        setState({state: 'error', error: error.message});
      }
    }
    
    fetchPlans();
  }, [variantId]);

  return state;
}
```

## GraphQL Query

Fetches selling plan data:

```graphql
query SellingPlansQuery($id: ID!) {
  productVariant(id: $id) {
    id
    title
    product {
      id
      title
    }
    image {
      url
      altText
    }
    sellingPlanGroups(first: 10) {
      edges {
        node {
          id
          name
          options
          sellingPlans(first: 10) {
            edges {
              node {
                id
                name
                description
                billingPolicy {
                  ... on SellingPlanRecurringBillingPolicy {
                    interval
                    intervalCount
                  }
                }
                deliveryPolicy {
                  ... on SellingPlanRecurringDeliveryPolicy {
                    interval
                    intervalCount
                  }
                }
                pricingPolicies {
                  ... on SellingPlanRecurringPricingPolicy {
                    adjustmentType
                    adjustmentValue {
                      ... on SellingPlanPricingPolicyPercentageValue {
                        percentage
                      }
                      ... on MoneyV2 {
                        amount
                        currencyCode
                      }
                    }
                  }
                }
              }
            }
          }
        }
      }
    }
  }
}
```

## Utilities

### Currency Formatting

```typescript
export function formatPricingPolicy(
  pricingPolicy: PricingPolicy | null
): [string, string] | null {
  if (!pricingPolicy) return null;
  
  const {adjustmentType, adjustmentValue} = pricingPolicy;
  
  if (adjustmentType === 'PERCENTAGE') {
    return ['percentage', `${adjustmentValue.percentage}%`];
  }
  
  if (adjustmentType === 'FIXED_AMOUNT') {
    return ['amount', `$${adjustmentValue.amount}`];
  }
  
  if (adjustmentType === 'PRICE') {
    return ['price', `$${adjustmentValue.amount}`];
  }
  
  return null;
}
```

### Version Comparison

```typescript
export function isVersionSupported(
  currentVersion: string,
  minimumVersion: string
): boolean {
  const current = currentVersion.split('.').map(Number);
  const minimum = minimumVersion.split('.').map(Number);
  
  for (let i = 0; i < 3; i++) {
    if (current[i] > minimum[i]) return true;
    if (current[i] < minimum[i]) return false;
  }
  
  return true;
}
```

### Cart Utilities

```typescript
export function getProductNameFromVariant(
  variant: ProductVariant,
  lineItem: LineItem
): string {
  return variant?.product?.title || lineItem.productTitle || '';
}

export function getVariantName(variant: ProductVariant): string {
  return variant?.title || '';
}

export function getVariantImage(
  variant: ProductVariant
): {url: string; altText: string} | undefined {
  return variant?.image
    ? {
        url: variant.image.url,
        altText: variant.image.altText || '',
      }
    : undefined;
}
```

## Best Practices

  <Accordion title="Performance">
