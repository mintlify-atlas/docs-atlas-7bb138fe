---
title: Admin Subscription Action Extension
description: Merchant-facing extension for creating and managing subscription plans in Shopify Admin
---

The Admin Subscription Action extension enables merchants to create and manage subscription selling plans directly from the Shopify Admin product and variant pages.

## Extension Configuration

Configured in `shopify.extension.toml`:

```toml
api_version = "unstable"

[[extensions]]
name = "t:name"
handle = "admin-subs-action"
type = "ui_extension"

[extensions.capabilities]
api_access = true
network_access = true

[[extensions.targeting]]
module = "./src/ProductExtension.tsx"
target = "admin.product-purchase-option.action.render"

[[extensions.targeting]]
module = "./src/ProductVariantExtension.tsx"
target = "admin.product-variant-purchase-option.action.render"
```

### Extension Targets

The extension renders at two surfaces:

<Tabs>
  <Tab title="Product Level">
    **Target**: `admin.product-purchase-option.action.render`
    
    Renders in the purchase options section of the product detail page. Allows merchants to create subscription plans that apply to all variants.

    ```tsx
    import {reactExtension} from '@shopify/ui-extensions-react/admin';
    import PurchaseOptionsActionExtension from './PurchaseOptionsActionExtension';
    import AdminExtensionContext from 'foundation/AdminExtensionContext';

    export default reactExtension(
      'admin.product-purchase-option.action.render',
      () => (
        <AdminExtensionContext.Provider value="PRODUCT">
          <PurchaseOptionsActionExtension />
        </AdminExtensionContext.Provider>
      )
    );
    ```
  </Tab>
  
  <Tab title="Variant Level">
    **Target**: `admin.product-variant-purchase-option.action.render`
    
    Renders in the purchase options section of individual variant pages. Enables variant-specific subscription plans.

    ```tsx
    import {reactExtension} from '@shopify/ui-extensions-react/admin';
    import PurchaseOptionsActionExtension from './PurchaseOptionsActionExtension';
    import AdminExtensionContext from 'foundation/AdminExtensionContext';

    export default reactExtension(
      'admin.product-variant-purchase-option.action.render',
      () => (
        <AdminExtensionContext.Provider value="PRODUCT_VARIANT">
          <PurchaseOptionsActionExtension />
        </AdminExtensionContext.Provider>
      )
    );
    ```
  </Tab>
</Tabs>

## Core Component

The main component provides a form for creating and editing subscription plans:

```tsx
import {
  AdminAction,
  BlockStack,
  Button,
  TextField,
  Checkbox,
  ChoiceList,
} from '@shopify/ui-extensions-react/admin';

export default function PurchaseOptionsActionExtension() {
  const {i18n, close, data} = useExtensionApi();
  const sellingPlanGroupId = data.selected[0]['sellingPlanId'];
  
  const [planName, setPlanName] = useState('');
  const [merchantCode, setMerchantCode] = useState('');
  const [offerDiscount, setOfferDiscount] = useState(true);
  const [discountType, setDiscountType] = useState('PERCENTAGE');
  const [deliveryOptions, setDeliveryOptions] = useState([]);

  async function handleSave() {
    // Create or update selling plan group
    if (sellingPlanGroupId) {
      await updateSellingPlanGroup({...});
    } else {
      await createSellingPlanGroup({...});
    }
    close();
  }

  return (
    <AdminAction
      primaryAction={<Button onPress={handleSave}>Save</Button>}
      secondaryAction={<Button onPress={close}>Cancel</Button>}
    >
      <BlockStack gap="base">
        <TextField
          label="Plan name"
          value={planName}
          onChange={setPlanName}
        />
        <TextField
          label="Merchant code"
          value={merchantCode}
          onChange={setMerchantCode}
        />
        <Checkbox
          checked={offerDiscount}
          onChange={setOfferDiscount}
        >
          Offer discount
        </Checkbox>
        {offerDiscount && (
          <ChoiceList
            choices={[
              {label: 'Percentage off', id: 'PERCENTAGE'},
              {label: 'Amount off', id: 'AMOUNT'},
              {label: 'Fixed price', id: 'PRICE'},
            ]}
            value={discountType}
            onChange={setDiscountType}
          />
        )}
        {/* Delivery options */}
      </BlockStack>
    </AdminAction>
  );
}
```

### Delivery Options

Merchants can add multiple delivery frequency options:

```tsx
class DeliveryOption {
  frequency: number;
  interval: 'DAY' | 'WEEK' | 'MONTH' | 'YEAR';
  discount: number;
  discountType: 'PERCENTAGE' | 'AMOUNT' | 'PRICE';
  
  toSellingPlanInput(currencyCode: string) {
    return {
      name: `Every ${this.frequency} ${this.interval.toLowerCase()}(s)`,
      billingPolicy: {
        recurring: {
          interval: this.interval,
          intervalCount: this.frequency,
        },
      },
      deliveryPolicy: {
        recurring: {
          interval: this.interval,
          intervalCount: this.frequency,
        },
      },
      pricingPolicies: this.discount ? [{
        recurring: {
          adjustmentType: this.discountType,
          adjustmentValue: {
            [this.discountType === 'PERCENTAGE' ? 'percentage' : 'fixedAmount']: 
              this.discountType === 'PERCENTAGE' 
                ? this.discount 
                : {amount: this.discount, currencyCode},
          },
        },
      }] : [],
    };
  }
}
```

## GraphQL Mutations

### Create Selling Plan Group

```graphql
mutation CreateSellingPlanGroup(
  $input: SellingPlanGroupInput!
  $resources: SellingPlanGroupResourceInput!
) {
  sellingPlanGroupCreate(
    input: $input
    resources: $resources
  ) {
    sellingPlanGroup {
      id
      name
      merchantCode
      sellingPlans(first: 10) {
        edges {
          node {
            id
            name
            billingPolicy {
              ... on SellingPlanRecurringBillingPolicy {
                interval
                intervalCount
              }
            }
            deliveryPolicy {
              ... on SellingPlanRecurringDeliveryPolicy {
                interval
                intervalCount
              }
            }
            pricingPolicies {
              ... on SellingPlanRecurringPricingPolicy {
                adjustmentType
                adjustmentValue {
                  ... on SellingPlanPricingPolicyPercentageValue {
                    percentage
                  }
                  ... on MoneyV2 {
                    amount
                    currencyCode
                  }
                }
              }
            }
          }
        }
      }
    }
    userErrors {
      field
      message
    }
  }
}
```

mutation UpdateSellingPlanGroup(

## Custom Hooks

### useSellingPlanGroupDetails

Fetches existing selling plan group data for editing:

```typescript
export function useSellingPlanGroupDetails({id}: {id?: string}) {
  const {graphql} = useExtensionApi();
  
  const {data, loading} = graphql(
    SellingPlanGroupDetailsQuery,
    {
      variables: {id},
      skip: !id,
    }
  );
  
  return {
    sellingPlanGroup: data?.sellingPlanGroup,
    loading,
  };
}
```

### useCreateSellingPlanGroup

Creates a new selling plan group:

```typescript
export function useCreateSellingPlanGroup() {
  const {graphql} = useExtensionApi();
  
  const [createMutation, {loading}] = graphql(
    CreateSellingPlanGroupMutation
  );
  
  async function createSellingPlanGroup({input, resources}) {
    const {data, errors} = await createMutation({
      variables: {input, resources},
    });
    
    return {data, errors};
  }
  
  return {
    createSellingPlanGroup,
    graphqlLoading: loading,
  };
}
```

### useUpdateSellingPlanGroup

Updates an existing selling plan group:

```typescript
export function useUpdateSellingPlanGroup() {
  const {graphql} = useExtensionApi();
  
  const [updateMutation, {loading}] = graphql(
    UpdateSellingPlanGroupMutation
  );
  
  async function updateSellingPlanGroup({id, input}) {
    const {data, errors} = await updateMutation({
      variables: {id, input},
    });
    
    return {data, errors};
  }
  
  return {
    updateSellingPlanGroup,
    graphqlLoading: loading,
  };
}
```

## Validation

The extension uses Zod for input validation:

```typescript
import {z} from 'zod';

export function getExtensionSellingPlanGroupValidator(translate) {
  return z.object({
    planName: z.string().min(1, translate('errors.planNameRequired')),
    merchantCode: z.string().min(1, translate('errors.merchantCodeRequired')),
    deliveryOptions: z.array(
      z.object({
        frequency: z.number().min(1),
        interval: z.enum(['DAY', 'WEEK', 'MONTH', 'YEAR']),
        discount: z.number().optional(),
      })
    ).min(1, translate('errors.deliveryOptionsRequired')),
  });
}
```

## Delivery Option Item Component

Displays individual delivery frequency options:

```tsx
export function DeliveryOptionItem({
  index,
  option,
  offerDiscount,
  discountType,
  updateDeliveryOption,
  removeDeliveryOption,
}) {
  return (
    <Card>
      <BlockStack gap="base">
        <InlineStack gap="small">
          <TextField
            label="Frequency"
            type="number"
            value={option.frequency}
            onChange={(value) => 
              updateDeliveryOption('frequency', value)
            }
          />
          <Select
            label="Interval"
            options={[
              {label: 'Days', value: 'DAY'},
              {label: 'Weeks', value: 'WEEK'},
              {label: 'Months', value: 'MONTH'},
              {label: 'Years', value: 'YEAR'},
            ]}
            value={option.interval}
            onChange={(value) => 
              updateDeliveryOption('interval', value)
            }
          />
        </InlineStack>
        {offerDiscount && (
          <TextField
            label={`Discount (${discountType})`}
            type="number"
            value={option.discount}
            onChange={(value) => 
              updateDeliveryOption('discount', value)
            }
          />
        )}
        {removeDeliveryOption && (
          <Button onPress={removeDeliveryOption} destructive>
            Remove
          </Button>
        )}
      </BlockStack>
    </Card>
  );
}
```
